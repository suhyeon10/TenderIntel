1. 카카오맵 Static Map API 사용
2. 구글 맵 Static Map API 사용
3. 수동으로 스크린샷 후 편집

const getMapImageUrl = (orgId: string): string => {

    moel_complaint: '/images/legal/organizations/moel-complaint-center-map.png',
    comwel: '/images/legal/organizations/comwel-daejeon-map.png',
    human_rights: '/images/legal/organizations/human-rights-commission-map.png',
    labor_attorney: '/images/legal/organizations/labor-attorney-generic-map.png',
    company_hr: '/images/legal/organizations/company-hr-generic-map.png',
    police: '/images/legal/organizations/police-station-generic-map.png',
    free_labor_consult: '/images/legal/organizations/free-labor-consult-map.png',
  }
  return imageMap[orgId] || '/images/legal/organizations/default-map.png'
}
